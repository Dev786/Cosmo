# Phase 0: Face Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Render a frameless, transparent, always-on-top Electron window (220×170px) with two animated eyes that blink and gaze-wander at 60fps while consuming less than 1% CPU at idle, with a tray toggle and persisted window position.

**Architecture:** The main process owns the BrowserWindow, tray, and electron-store position persistence; a preload script exposes a typed `cosmo` bridge over contextBridge; the renderer loads the classic ExpressionPack from the pack registry and drives animation entirely in the renderer via requestAnimationFrame — the main process never drives animation frames. The ExpressionPack contract (`init / setState / pulse / setIntensity / setActivity / dispose`) is the only boundary between app logic and rendering code, keeping the two wholly independent.

**Tech Stack:** Electron (latest stable), TypeScript 5 (strict), electron-store 10, zod 3, electron-builder (dev), Jest + ts-jest (tests), native `fs/path/os` (logger — no extra deps).

```bash
# We use a tiny Python script to generate a placeholder PNG
python3 -c "
import struct, zlib

def create_png(size, color):
    def chunk(ctype, data):
        c = struct.pack('>I', len(data)) + ctype + data
        return c + struct.pack('>I', zlib.crc32(c[4:]) & 0xffffffff)
    
    header = b'\\x89PNG\\r\\n\\x1a\\n'
    ihdr_data = struct.pack('>IIBBBBB', size, size, 8, 2, 0, 0, 0)
    ihdr = chunk(b'IHDR', ihdr_data)
    
    raw = b''
    for _ in range(size):
        row = b'\\x00' + bytes(color) * size
        raw += row
    
    idat = chunk(b'IDAT', zlib.compress(raw))
    iend = chunk(b'IEND', b'')
    
    return header + ihdr + idat + iend

# 16x16 white circle on transparent... simplified: just white square
with open('assets/tray-icon.png', 'wb') as f:
    f.write(create_png(16, [255, 255, 255]))

with open('assets/tray-icon@2x.png', 'wb') as f:
    f.write(create_png(32, [255, 255, 255]))
```

```typescript
// src/main/core/log.ts
// Rotating daily file logger that writes to ~/.pixel/logs/ (or override via opts).
// Exported _pruneOldLogs is for testing and the daily rotation check.
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';
import type { Logger } from '../../shared/types';

export interface LoggerOptions {
  /** Override log directory (useful for tests). Defaults to ~/.pixel/logs */
  logDir?: string;
  /** Maximum number of daily log files to keep. Defaults to 7 */
  retentionDays?: number;
}

type LogLevel = 'INFO' | 'WARN' | 'ERROR' | 'DEBUG';

function defaultLogDir(): string {
  return path.join(os.homedir(), '.pixel', 'logs');
}

function todayFileName(): string {
  return `cosmo-${new Date().toISOString().slice(0, 10)}.log`;
}

function serializeExtra(args: unknown[]): string {
  if (args.length === 0) return '';
  return ' ' + args.map(a => {
    if (a instanceof Error) return `${a.name}: ${a.message}`;
    if (typeof a === 'object') {
      try { return JSON.stringify(a); } catch { return String(a); }
    }
    return String(a);
  }).join(' ');
}

/**
 * Prune log files older than `keepDays` from `logDir`.
 * Exported so tests can call it directly without running the full logger.
 */
export function _pruneOldLogs(logDir: string, keepDays: number): void {
  try {
    const files = fs.readdirSync(logDir)
      .filter(f => /^cosmo-\d{4}-\d{2}-\d{2}\.log$/.test(f))
      .sort(); // ISO dates sort lexicographically

    const excess = files.length - keepDays;
    for (let i = 0; i < excess; i++) {
      fs.unlinkSync(path.join(logDir, files[i]));
    }
  } catch {
    // Non-fatal: if pruning fails, we just accumulate an extra file
  }
}

export function createLogger(opts: LoggerOptions = {}): Logger {
  const logDir = opts.logDir ?? defaultLogDir();
  const retentionDays = opts.retentionDays ?? 7;

  // Ensure directory exists
  fs.mkdirSync(logDir, { recursive: true });

  // Prune on creation (once per process start)
  _pruneOldLogs(logDir, retentionDays);

  let currentDate = new Date().toISOString().slice(0, 10);
  let currentPath = path.join(logDir, todayFileName());
  let stream = fs.createWriteStream(currentPath, { flags: 'a' });

  function write(level: LogLevel, msg: string, extra: unknown[]): void {
    // Roll file at midnight
    const today = new Date().toISOString().slice(0, 10);
    if (today !== currentDate) {
      stream.end();
      currentDate = today;
      currentPath = path.join(logDir, `cosmo-${today}.log`);
      stream = fs.createWriteStream(currentPath, { flags: 'a' });
      _pruneOldLogs(logDir, retentionDays);
    }

    const entry = JSON.stringify({
      ts: new Date().toISOString(),
      level: `[${level}]`,
      msg,
      extra: extra.length > 0 ? serializeExtra(extra) : undefined,
    });

    stream.write(entry + '\n');

    // Mirror to console in development
    if (process.env['NODE_ENV'] !== 'production') {
      const prefix = `[${level}] ${msg}`;
      if (level === 'ERROR') console.error(prefix + serializeExtra(extra));
      else if (level === 'WARN') console.warn(prefix + serializeExtra(extra));
      else console.log(prefix + serializeExtra(extra));
    }
  }

  return {
    info(msg, ...args) { write('INFO', msg, args); },
    warn(msg, ...args) { write('WARN', msg, args); },
    error(msg, ...args) { write('ERROR', msg, args); },
    debug(msg, ...args) { write('DEBUG', msg, args); },
  };
}

// ---------------------------------------------------------------------------
```

```typescript
// Loads the requested ExpressionPack by name.
// Falls back to 'classic' with a console.warn if the name is unknown.
// Adding a new pack = adding its folder and one import line here.
import type { ExpressionPack } from './types';
import { ClassicPack } from './classic/index';

const PACK_MAP: Record<string, () => ExpressionPack> = {
  classic: () => new ClassicPack(),
};

/**
 * Load and initialize an ExpressionPack.
 * @param name - Pack name from config (e.g. 'classic')
 * @param container - The DOM element the pack should render into
 * @param reducedMotion - Whether the user prefers reduced motion
 * @returns An initialized ExpressionPack instance
 */
export function loadPack(
  name: string,
  container: HTMLElement,
  reducedMotion: boolean,
): ExpressionPack {
  const factory = PACK_MAP[name];

  if (!factory) {
    console.warn(
      `[packs/registry] Unknown expression pack "${name}" — falling back to "classic". ` +
      `Valid options: ${Object.keys(PACK_MAP).join(', ')}`,
    );
    const fallback = new ClassicPack();
    fallback.init(container, { reducedMotion });
    return fallback;
  }

  const pack = factory();
  pack.init(container, { reducedMotion });
  return pack;
}
```

**Design notes (from research — implement these exactly):**

The eye structure is layered: an outer `.eye` div (the white sclera), an `.eye-inner` div (the colored iris, ~70% of sclera size), a `.pupil` div (dark, ~60% of iris size, shifted slightly upward for an innocent look), and a `.highlight` dot (white, ~8px, top-right of pupil). This layered structure is what creates the "moe/kawaii" look — the ratio of iris-to-sclera and the highlight dot placement are the two most critical factors. See Task 7.5 (Cute Face Details) and Task 8 for the DOM structure.

Key design principles applied here:
- **Circular eyes for cuteness**: large `border-radius` (50%) creates round eyes. Moe designs use ~70% iris coverage.
- **Soft warm color**: `#F5F5F5` sclera (near-white, not pure white — pure white feels harsh), `#5B8DD9` iris (soft blue — associated with innocence and openness), `#1A1A2E` pupil (near-black with a slight blue tint).
- **Innocent pupil offset**: pupil shifted 2px upward (looking slightly "up") — reinforces innocent, curious expression.
- **Happy = ∩ arc**: `border-radius` is set to make the bottom corners round and top corners sharp — achieved with `border-radius: 50% 50% 0 0` style. The `h` is very small to create the anime crescent.
- **Bored = half-lidded**: the top eyelid div descends 40% from the top — parametrized as `eyelidTopPct` below.
- **Annoyed = flat top**: `border-radius` becomes asymmetric — the top edge is flatter (4px) and the bottom stays round (12px), giving the "pressed-down" angry lid look.
- **Sleeping = thin curved line**: 3px height, maximum `border-radius`, soft grey.

```typescript
// src/renderer/packs/classic/tween.ts
// Lightweight tweening utilities for eye shape parameters.
// Uses requestAnimationFrame for smooth 60fps animation.
// No external dependencies.
import type { EyeParams } from './recipes';

// ---------------------------------------------------------------------------
// Math helpers
// ---------------------------------------------------------------------------

/** Linear interpolation — no clamping, callers may supply t outside [0,1] */
export function lerp(a: number, b: number, t: number): number {
  return a + (b - a) * t;
}

/**
 * Cubic ease-in-out (smoothstep-style).
 * t must be in [0, 1].
 */
function easeInOut(t: number): number {
  return t < 0.5
    ? 4 * t * t * t
    : 1 - Math.pow(-2 * t + 2, 3) / 2;
}

// ---------------------------------------------------------------------------
// Color interpolation (hex #rrggbb only)
// ---------------------------------------------------------------------------

function hexToRgb(hex: string): [number, number, number] {
  const h = hex.replace('#', '');
  const n = parseInt(h, 16);
  return [(n >> 16) & 0xff, (n >> 8) & 0xff, n & 0xff];
}

function rgbToHex(r: number, g: number, b: number): string {
  return '#' + [r, g, b].map(v => Math.round(v).toString(16).padStart(2, '0')).join('');
}

function lerpColor(from: string, to: string, t: number): string {
  const [r1, g1, b1] = hexToRgb(from);
  const [r2, g2, b2] = hexToRgb(to);
  return rgbToHex(lerp(r1, r2, t), lerp(g1, g2, t), lerp(b1, b2, t));
}

// ---------------------------------------------------------------------------
// Main tween function
// ---------------------------------------------------------------------------

/**
 * Tween from `from` EyeParams to `to` EyeParams over `duration` milliseconds.
 *
 * @param from     Starting eye parameters
 * @param to       Target eye parameters
 * @param duration Animation duration in milliseconds
 * @param onUpdate Called each animation frame with interpolated params
 * @param onComplete Called once when the tween finishes
 * @returns A cancel function — call it to stop the tween immediately
 */
export function tweenParams(
  from: EyeParams,
  to: EyeParams,
  duration: number,
  onUpdate: (p: EyeParams) => void,
  onComplete: () => void,
): () => void {
  let cancelled = false;
  let startTime: number | null = null;
  let rafId: number;

  function frame(now: number) {
    if (cancelled) return;

    if (startTime === null) {
      startTime = now;
    }

    const elapsed = now - startTime;
    const rawT = Math.min(elapsed / duration, 1);
    const t = easeInOut(rawT);

    const current: EyeParams = {
      w: lerp(from.w, to.w, t),
      h: lerp(from.h, to.h, t),
      radius: lerp(from.radius, to.radius, t),
      offsetX: lerp(from.offsetX, to.offsetX, t),
      offsetY: lerp(from.offsetY, to.offsetY, t),
      rotation: lerp(from.rotation, to.rotation, t),
      color: lerpColor(from.color, to.color, t),
    };

    onUpdate(current);

    if (rawT >= 1) {
      // Snap to exact target to avoid floating-point drift
      onUpdate({ ...to });
      onComplete();
      return;
    }

    rafId = requestAnimationFrame(frame);
  }

  rafId = requestAnimationFrame(frame);

  return () => {
    cancelled = true;
    cancelAnimationFrame(rafId);
  };
}
```

```typescript
// Classic expression pack: two rounded-rect eye divs driven by CSS transforms.
// Implements the ExpressionPack contract. Logic here; numbers in recipes.ts.
import type { ExpressionPack, MoodState, ActivityState, PulseEvent } from '../types';
import { RECIPES, type EyeParams } from './recipes';
import { tweenParams } from './tween';

// Gap between the two eyes in pixels
const EYE_GAP = 28;

// Gaze wander amplitude (px) and timing
const GAZE_AMPLITUDE = 12;
const GAZE_MIN_INTERVAL = 2000;
const GAZE_INTERVAL_RANGE = 2000;

// Blink timing
const BLINK_MIN_INTERVAL = 3000;
const BLINK_INTERVAL_RANGE = 3000;
const BLINK_DURATION = 130;

// LookAway duration (ms) and offset
const LOOK_AWAY_OFFSET = 60;
const LOOK_AWAY_HOLD = 2000;
const LOOK_AWAY_TWEEN = 200;

// ---------------------------------------------------------------------------
// DOM helpers
// ---------------------------------------------------------------------------

function applyParams(el: HTMLElement, params: EyeParams, gazeOffsetX = 0, gazeOffsetY = 0): void {
  const totalX = params.offsetX + gazeOffsetX;
  const totalY = params.offsetY + gazeOffsetY;
  el.style.width = `${params.w}px`;
  el.style.height = `${params.h}px`;
  el.style.borderRadius = `${params.radius}px`;
  el.style.backgroundColor = params.color;
  el.style.transform = `translate(${totalX}px, ${totalY}px) rotate(${params.rotation}deg)`;
}

function createEyeEl(): HTMLElement {
  const el = document.createElement('div');
  el.className = 'eye';
  el.style.cssText = `
    position: absolute;
    top: 50%;
    transition: none;
    will-change: transform, width, height;
    flex-shrink: 0;
  `;
  return el;
}

export class ClassicPack implements ExpressionPack {
  readonly name = 'classic';

  // DOM
  private container: HTMLElement | null = null;
  private leftEye: HTMLElement | null = null;
  private rightEye: HTMLElement | null = null;

  // State
  private currentParams: EyeParams = { ...RECIPES.idle };
  private currentMood: MoodState = 'idle';
  private reducedMotion = false;

  // Gaze wander state
  private gazeX = 0;
  private gazeY = 0;

  // Active tween cancel fn
  private cancelTween: (() => void) | null = null;

  // Timers
  private gazeTimer: ReturnType<typeof setTimeout> | null = null;
  private blinkTimer: ReturnType<typeof setTimeout> | null = null;

  // Intensity state (0..1 for annoyed color ramp)
  private intensity = 0;

  // Activity overlay (stub — Phase 1 will add rich overlays)
  private activityEl: HTMLElement | null = null;

  // ---------------------------------------------------------------------------
  // ExpressionPack interface
  // ---------------------------------------------------------------------------

  init(container: HTMLElement, opts: { reducedMotion: boolean }): void {
    if (this.container) this.dispose(); // re-init safety

    this.container = container;
    this.reducedMotion = opts.reducedMotion;

    // Create wrapper for both eyes
    const wrapper = document.createElement('div');
    wrapper.style.cssText = `
      position: relative;
      display: flex;
      flex-direction: row;
      align-items: center;
      justify-content: center;
      gap: ${EYE_GAP}px;
      width: 100%;
      height: 100%;
    `;

    this.leftEye = createEyeEl();
    this.rightEye = createEyeEl();

    // Mirror right eye: slight inward rotation for personality
    this.rightEye.style.transform = 'scaleX(-1)';

    wrapper.appendChild(this.leftEye);
    wrapper.appendChild(this.rightEye);
    container.appendChild(wrapper);

    // Apply initial idle recipe
    this.applyToBothEyes(this.currentParams);

    // Start timers unless reduced motion
    if (!this.reducedMotion) {
      this.scheduleGaze();
      this.scheduleBlink();
    } else {
      // Still blink even with reduced motion (it's not just motion — it's a state)
      this.scheduleBlink();
    }
  }

  setState(state: MoodState): void {
    if (!this.leftEye || !this.rightEye) return;

    this.currentMood = state;
    const target = this.effectiveTarget(state);

    // Cancel any in-progress tween
    this.cancelTween?.();

    this.cancelTween = tweenParams(
      this.currentParams,
      target,
      250,
      (p) => {
        this.currentParams = p;
        this.applyToBothEyes(p);
      },
      () => {
        this.cancelTween = null;
      },
    );
  }

  pulse(event: PulseEvent): void {
    if (!this.leftEye || !this.rightEye) return;

    switch (event) {
      case 'blink':
        this.doBlink();
        break;
      case 'lookAway':
        this.doLookAway();
        break;
      case 'speakTick':
        // Slight vertical pulse — handled by speaking state tween
        break;
      case 'soundBloop':
      case 'soundChime':
      case 'soundGrumble':
        // Visual audio feedback — stub for Phase 1
        break;
    }

    // level: 0..1
    // At 0: use the state's normal color
    // At >0: interpolate annoyed color toward deeper red
    this.intensity = Math.max(0, Math.min(1, level));

    if (this.currentMood === 'annoyed' && this.leftEye && this.rightEye) {
      const baseColor = RECIPES.annoyed.color;           // '#f5a623' (amber)
      const maxColor = '#e74c3c';                         // red
      const color = this.lerpHexColor(baseColor, maxColor, this.intensity);
      this.leftEye.style.backgroundColor = color;
      this.rightEye.style.backgroundColor = color;
    }
  }

  setActivity(activity: ActivityState | null): void {
    // Phase 0 stub: activity overlays are not rendered yet.
    // Phase 1 will add music equalizer, radar dot, and timer ring.
    // We store the value so the overlay logic can use it when added.
    void activity; // intentional no-op for now

  dispose(): void {
    // Cancel tweens and timers
    this.cancelTween?.();
    this.cancelTween = null;

    if (this.gazeTimer !== null) {
      clearTimeout(this.gazeTimer);
      this.gazeTimer = null;
    }
    if (this.blinkTimer !== null) {
      clearTimeout(this.blinkTimer);
      this.blinkTimer = null;
    }

    // Remove DOM elements
    if (this.container) {
      this.container.innerHTML = '';
    }

    this.leftEye = null;
    this.rightEye = null;
    this.container = null;
  }

  // ---------------------------------------------------------------------------
  // Private helpers
  // ---------------------------------------------------------------------------

  private effectiveTarget(state: MoodState): EyeParams {
    const base = { ...RECIPES[state] };
    // Preserve any active intensity color if already annoyed
    if (state === 'annoyed' && this.intensity > 0) {
      base.color = this.lerpHexColor(base.color, '#e74c3c', this.intensity);
    }
    return base;
  }

  private applyToBothEyes(params: EyeParams): void {
    if (!this.leftEye || !this.rightEye) return;
    // Left eye: direct
    applyParams(this.leftEye, params, this.gazeX, this.gazeY);
    // Right eye: mirrored horizontal gaze, same vertical
    applyParams(this.rightEye, params, -this.gazeX, this.gazeY);
  }

  private scheduleGaze(): void {
    const delay = GAZE_MIN_INTERVAL + Math.random() * GAZE_INTERVAL_RANGE;
    this.gazeTimer = setTimeout(() => {
      this.doGazeWander();
      this.scheduleGaze();
    }, delay);
  }

  private doGazeWander(): void {
    if (this.reducedMotion || this.currentMood === 'sleeping') return;

    const targetX = (Math.random() * 2 - 1) * GAZE_AMPLITUDE;
    const targetY = (Math.random() * 2 - 1) * GAZE_AMPLITUDE * 0.5;

    const startX = this.gazeX;
    const startY = this.gazeY;
    const duration = 600;
    let startTime: number | null = null;
    let rafId: number;

    const animate = (now: number) => {
      if (startTime === null) startTime = now;
      const t = Math.min((now - startTime) / duration, 1);
      const ease = t < 0.5 ? 4 * t * t * t : 1 - Math.pow(-2 * t + 2, 3) / 2;
      this.gazeX = startX + (targetX - startX) * ease;
      this.gazeY = startY + (targetY - startY) * ease;
      this.applyToBothEyes(this.currentParams);
      if (t < 1) rafId = requestAnimationFrame(animate);
    };
    rafId = requestAnimationFrame(animate);
  }

  private scheduleBlink(): void {
    const delay = BLINK_MIN_INTERVAL + Math.random() * BLINK_INTERVAL_RANGE;
    this.blinkTimer = setTimeout(() => {
      this.doBlink();
      this.scheduleBlink();
    }, delay);
  }

  private doBlink(): void {
    if (!this.leftEye || !this.rightEye) return;

    // Store original scaleY (may be overridden by setState)
    const left = this.leftEye;
    const right = this.rightEye;
    const prevLeftTransform = left.style.transform;
    const prevRightTransform = right.style.transform;

    // Close
    left.style.transition = `transform ${BLINK_DURATION / 2}ms ease-in`;
    right.style.transition = `transform ${BLINK_DURATION / 2}ms ease-in`;

    const applyBlink = (el: HTMLElement, baseTransform: string, closed: boolean) => {
      // Replace any existing scaleY in the transform string, or append it
      const withoutScale = baseTransform.replace(/\s*scaleY\([^)]+\)/g, '');
      el.style.transform = closed
        ? `${withoutScale} scaleY(0.08)`
        : withoutScale;
    };

    applyBlink(left, prevLeftTransform, true);
    applyBlink(right, prevRightTransform, true);

    // Open after half-duration
    setTimeout(() => {
      if (!this.leftEye || !this.rightEye) return;
      left.style.transition = `transform ${BLINK_DURATION / 2}ms ease-out`;
      right.style.transition = `transform ${BLINK_DURATION / 2}ms ease-out`;
      applyBlink(left, left.style.transform, false);
      applyBlink(right, right.style.transform, false);

      // Clear transition after blink completes
      setTimeout(() => {
        if (left) left.style.transition = 'none';
        if (right) right.style.transition = 'none';
      }, BLINK_DURATION / 2 + 10);
    }, BLINK_DURATION / 2);
  }

  private doLookAway(): void {
    if (!this.leftEye || !this.rightEye) return;

    const startX = this.gazeX;
    const startY = this.gazeY;

    // Animate gaze to LOOK_AWAY_OFFSET
    let startTime: number | null = null;
    const lookAwayTween = (now: number) => {
      if (startTime === null) startTime = now;
      const t = Math.min((now - startTime) / LOOK_AWAY_TWEEN, 1);
      this.gazeX = startX + (LOOK_AWAY_OFFSET - startX) * t;
      this.applyToBothEyes(this.currentParams);
      if (t < 1) requestAnimationFrame(lookAwayTween);
    };
    requestAnimationFrame(lookAwayTween);

    // Return after hold
    setTimeout(() => {
      let start2: number | null = null;
      const returnTween = (now: number) => {
        if (start2 === null) start2 = now;
        const t = Math.min((now - start2) / LOOK_AWAY_TWEEN, 1);
        this.gazeX = LOOK_AWAY_OFFSET + (startX - LOOK_AWAY_OFFSET) * t;
        this.applyToBothEyes(this.currentParams);
        if (t < 1) requestAnimationFrame(returnTween);
      };
      requestAnimationFrame(returnTween);
    }, LOOK_AWAY_HOLD);
  }

  private lerpHexColor(from: string, to: string, t: number): string {
    const hexToRgb = (hex: string): [number, number, number] => {
      const h = hex.replace('#', '');
      const n = parseInt(h, 16);
      return [(n >> 16) & 0xff, (n >> 8) & 0xff, n & 0xff];
    };
    const [r1, g1, b1] = hexToRgb(from);
    const [r2, g2, b2] = hexToRgb(to);
    const lerp = (a: number, b: number) => Math.round(a + (b - a) * t);
    const toHex = (n: number) => n.toString(16).padStart(2, '0');
    return `#${toHex(lerp(r1, r2))}${toHex(lerp(g1, g2))}${toHex(lerp(b1, b2))}`;
  }
}
```

### Task 7.5: Cute Face Details — Layered Eye DOM + CSS

**Why this task exists:** The layered inner structure (iris, pupil, highlight, eyelid, blush) is what makes the eyes read as cute rather than just functional geometric shapes. This is the most design-sensitive task in M0. Implement it carefully — all these sub-elements are rendered inside the `.eye` divs created by ClassicPack in Task 7.

**Files:**
- Modify: `src/renderer/packs/classic/index.ts` (add sub-element creation and state-driven updates)

**Design research summary:**

From moe/kawaii character design principles:
- The **iris-to-sclera ratio** (~70%) is the single most important factor for cute eyes. At 50%, eyes look normal; at 70%+, they read as "big-eyed" and cute.
- The **highlight dot** (white, ~8px, positioned top-right of the pupil) creates the "glossy" or "wet" look that makes eyes appear lively and emotional. Without it, eyes look flat.
- The **pupil positioned slightly upward** within the iris (a few px negative offsetY) creates an innocent, childlike "looking up" expression — the basis of the "moe gaze."
- **Blush marks** (pale pink ovals, 20–30px wide, 8–12px tall, positioned below and outside the eyes, semi-transparent) are the shorthand for happy/warm emotion in kawaii character design (think Tamagotchi, Cozmo in happy mode, nearly all VTuber avatars).
- A **subtle drop shadow** on the eye containers (`filter: drop-shadow(0 2px 4px rgba(0,0,0,0.12))`) separates them from any background and adds depth without being harsh.
- The **eyelid overlay** descends from the top of the eye as a separate div rather than `scaleY` — this looks more anatomically correct because the lid covers the top portion with a straight horizontal edge, not a vertical squish from the center.

**DOM structure for each eye (to be created inside `createEyeEl()`):**

```
.eye                      ← sclera: white/cream outer circle, has drop-shadow
  .eye-inner              ← iris: colored circle, 70% of sclera size

      .highlight          ← white dot, 8px, top-right of pupil (absolute)
  .eyelid-top             ← lid that descends from top (absolute, same bg as window = transparent/dark)
.blush-left               ← sibling outside .eye, below-left, pink oval
.blush-right              ← sibling outside .eye, below-right, pink oval
```

```typescript
// src/renderer/main.ts
// Renderer process entry point.
// Loads the expression pack and wires IPC events from the main process.
// Never imports from src/main/ — all communication is via the preload bridge.

import { loadPack } from './packs/registry';
import type { MoodState, ActivityState, PulseEvent } from '../shared/types';
import { IPC } from '../shared/types';

// ---------------------------------------------------------------------------
// Global type augmentation for the contextBridge-exposed API
// ---------------------------------------------------------------------------

declare global {
  interface Window {
    cosmo: {
      on(channel: string, cb: (...args: unknown[]) => void): void;
      invoke(channel: string, data?: unknown): Promise<unknown>;
      send(channel: string, data?: unknown): void;
    };
  }
}

// ---------------------------------------------------------------------------
// Boot
// ---------------------------------------------------------------------------

function boot(): void {
  const eyesContainer = document.getElementById('eyes-container');
  const micDot = document.getElementById('mic-dot');
  const chatArea = document.getElementById('chat-area');
  const chatText = document.getElementById('chat-text');
  const chatInput = document.getElementById('chat-input') as HTMLInputElement | null;

  if (!eyesContainer) {
    console.error('[renderer] #eyes-container not found — aborting pack init');
    return;
  }

  // Check prefers-reduced-motion
  const reducedMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;

  // Load and initialize the expression pack
  // Phase 0 always loads 'classic'. Later phases will read config.expressionPack.
  const pack = loadPack('classic', eyesContainer, reducedMotion);

  // ---------------------------------------------------------------------------
  // IPC event handlers (main → renderer)
  // ---------------------------------------------------------------------------

  window.cosmo.on(IPC.MOOD_SET, (state: unknown) => {
    pack.setState(state as MoodState);
  });

  window.cosmo.on(IPC.MOOD_PULSE, (event: unknown) => {
    pack.pulse(event as PulseEvent);
  });

  window.cosmo.on(IPC.MOOD_INTENSITY, (level: unknown) => {
    pack.setIntensity(level as number);
  });

  window.cosmo.on(IPC.ACTIVITY_SET, (activity: unknown) => {
    pack.setActivity(activity as ActivityState | null);
  });

  window.cosmo.on(IPC.CHAT_MESSAGE, (msg: unknown) => {
    if (!chatArea || !chatText) return;
    chatText.textContent = String(msg ?? '');
    chatArea.classList.add('visible');
  });

  // ---------------------------------------------------------------------------
  // Mic dot interaction
  // ---------------------------------------------------------------------------

  if (micDot) {
    // Click: trigger voice input
    micDot.addEventListener('click', () => {
      window.cosmo.send('voice:trigger');
    });

    // Keyboard accessibility
    micDot.addEventListener('keydown', (e: KeyboardEvent) => {
      if (e.key === 'Enter' || e.key === ' ') {
        e.preventDefault();
        window.cosmo.send('voice:trigger');
      }
    });
  }

  // ---------------------------------------------------------------------------
  // Chat input
  // ---------------------------------------------------------------------------

  if (chatInput) {
    // Show chat area when input is focused
    chatInput.addEventListener('focus', () => {
      chatArea?.classList.add('visible');
    });

    chatInput.addEventListener('keydown', (e: KeyboardEvent) => {
      if (e.key === 'Enter' && chatInput.value.trim()) {
        window.cosmo.send(IPC.CHAT_SUBMIT, chatInput.value.trim());
        chatInput.value = '';
      }
      if (e.key === 'Escape') {
        chatInput.blur();
      }
    });
  }

  // ---------------------------------------------------------------------------
  // Voice unavailability: main process can signal to keep mic dot visible
  // ---------------------------------------------------------------------------

  window.cosmo.on('voice:unavailable', () => {
    micDot?.classList.add('always-visible');
  });

  console.log('[renderer] Cosmo renderer initialized, pack:', pack.name);
}

// Wait for DOM
if (document.readyState === 'loading') {
  document.addEventListener('DOMContentLoaded', boot);
} else {
  boot();
}
```

```typescript
// src/main/index.ts
// Main process entry point.
// Owns: BrowserWindow creation, Tray, position persistence, IPC handlers,
// single-instance lock.
import * as path from 'path';
import { app, BrowserWindow, Tray, Menu, ipcMain, nativeImage, screen } from 'electron';
import Store from 'electron-store';
import type { Config } from '../shared/types';
import { CONFIG_DEFAULTS, IPC } from '../shared/types';
import { getLogger } from './core/log';

// ---------------------------------------------------------------------------
// Types
// ---------------------------------------------------------------------------

interface WindowBounds {
  x: number;
  y: number;
  width: number;
  height: number;
}

interface StoreSchema {
  windowBounds: WindowBounds;
  config: Config;
}

// ---------------------------------------------------------------------------
// Store factory (exported for tests)
// ---------------------------------------------------------------------------

export function createStore(): Store<StoreSchema> {
  return new Store<StoreSchema>({
    name: 'cosmo',
    cwd: path.join(require('os').homedir(), '.pixel'),
    defaults: {
      windowBounds: { x: -1, y: -1, width: 220, height: 170 },
      config: CONFIG_DEFAULTS,
    },
  });
}

// ---------------------------------------------------------------------------
// Window factory (exported for tests)
// ---------------------------------------------------------------------------

export function createWindow(
  store?: Store<StoreSchema>,
): BrowserWindow {
  const log = getLogger();

  const savedBounds = store?.get('windowBounds') as WindowBounds | undefined;

  // Default: top-right corner of primary display
  const primaryDisplay = screen.getPrimaryDisplay();
  const { width: screenW, height: _screenH } = primaryDisplay.workArea;
  const defaultX = screenW - 220 - 20;
  const defaultY = 40;

  const x = savedBounds && savedBounds.x >= 0 ? savedBounds.x : defaultX;
  const y = savedBounds && savedBounds.y >= 0 ? savedBounds.y : defaultY;

  const win = new BrowserWindow({
    width: 220,
    height: 170,
    x,
    y,
    frame: false,
    transparent: true,
    alwaysOnTop: true,
    skipTaskbar: true,
    resizable: false,
    hasShadow: false,
    webPreferences: {
      contextIsolation: true,
      nodeIntegration: false,
      preload: path.join(__dirname, 'preload.js'),
    },
  });

  // Load the renderer
  win.loadFile(path.join(__dirname, '../../src/renderer/index.html'));

  // Save position on move
  win.on('moved', () => {
    if (store) {
      const bounds = win.getBounds();
      store.set('windowBounds', bounds);
    }
  });

  log.info(`Window created at (${x}, ${y})`);
  return win;
}

// ---------------------------------------------------------------------------
// Tray factory (exported for tests)
// ---------------------------------------------------------------------------

export function createTray(win?: BrowserWindow): Tray {
  // Use a tiny blank icon; real icon replaced by assets/ at build time
  const icon = nativeImage.createEmpty().resize({ width: 16, height: 16 });

  const tray = new Tray(icon);
  tray.setToolTip('Cosmo');

  const buildMenu = () => Menu.buildFromTemplate([
    {
      label: win?.isVisible() ? 'Hide Cosmo' : 'Show Cosmo',
      click: () => {
        if (win) {
          if (win.isVisible()) win.hide();
          else win.show();
        }
        // Rebuild to update label
        tray.setContextMenu(buildMenu());
      },
    },
    { type: 'separator' },
    {
      label: 'Quit',
      click: () => app.quit(),
    },
  ]);

  tray.setContextMenu(buildMenu());

  // macOS: left-click also opens the menu
  tray.on('click', () => tray.popUpContextMenu());

  return tray;
}

// ---------------------------------------------------------------------------
// IPC handlers
// ---------------------------------------------------------------------------

function registerIPCHandlers(store: Store<StoreSchema>): void {
  ipcMain.handle(IPC.SETTINGS_GET, () => {
    return store.get('config');
  });

  ipcMain.handle(IPC.SETTINGS_SET, (_event, newConfig: Partial<Config>) => {
    const current = store.get('config') as Config;
    store.set('config', { ...current, ...newConfig });
    return store.get('config');
  });
}

// ---------------------------------------------------------------------------
// App lifecycle
// ---------------------------------------------------------------------------

function main(): void {
  const log = getLogger();

  // Single-instance lock
  const gotLock = app.requestSingleInstanceLock();
  if (!gotLock) {
    log.warn('Another instance is already running. Quitting.');
    app.quit();
    return;
  }

  // Hide from dock (always-on-top tool, not a dock app)
  app.dock?.hide();

  const store = createStore();

  app.whenReady().then(() => {
    const win = createWindow(store);
    const _tray = createTray(win);

    registerIPCHandlers(store);

    log.info('Cosmo started');

    // macOS: re-create window if all windows are closed but app is still active
    app.on('activate', () => {
      if (BrowserWindow.getAllWindows().length === 0) {
        createWindow(store);
      }
    });
  }).catch((err) => {
    log.error('Failed to initialize app', err);
    app.quit();
  });

  app.on('window-all-closed', () => {
    // On macOS, keep app alive unless explicitly quit
    if (process.platform !== 'darwin') {
      app.quit();
    }
  });

  // Global error handler
  process.on('uncaughtException', (err) => {
    log.error('Uncaught exception in main process', err);
  });

  process.on('unhandledRejection', (reason) => {
    log.error('Unhandled promise rejection in main process', reason);
  });
}

// Only run main() when this is the entry point, not when imported in tests
if (require.main === module) {
  main();
}
```

```typescript
// src/main/preload.ts
// Preload script: exposes a minimal, typed IPC bridge to the renderer.
// Allowlist of valid channels prevents arbitrary IPC from renderer code.
// contextIsolation: true ensures renderer JS cannot access Node APIs.
import { contextBridge, ipcRenderer } from 'electron';
import { IPC } from '../shared/types';

// All channels the renderer is allowed to receive from main
const RECEIVE_CHANNELS: string[] = [
  IPC.MOOD_SET,
  IPC.MOOD_PULSE,
  IPC.MOOD_INTENSITY,
  IPC.ACTIVITY_SET,
  IPC.CHAT_MESSAGE,
  // Voice unavailability notification
  'voice:unavailable',
  // Future: add channels here as features grow
];

// All channels the renderer is allowed to send to main
const SEND_CHANNELS: string[] = [
  IPC.CHAT_SUBMIT,
  IPC.SETTINGS_GET,
  IPC.SETTINGS_SET,
  // Voice trigger from mic dot click
  'voice:trigger',
];

// Channels that the renderer can invoke (request/response pattern)
const INVOKE_CHANNELS: string[] = [
  IPC.SETTINGS_GET,
  IPC.SETTINGS_SET,
];

contextBridge.exposeInMainWorld('cosmo', {
  /**
   * Subscribe to a channel from the main process.
   * Only channels in RECEIVE_CHANNELS are accepted.
   */
  on(channel: string, cb: (...args: unknown[]) => void): void {
    if (!RECEIVE_CHANNELS.includes(channel)) {
      console.warn(`[preload] Blocked unauthorized receive channel: ${channel}`);
      return;
    }
    // Wrap callback to strip the ipcRenderer Event object (first arg)
    ipcRenderer.on(channel, (_event, ...args) => cb(...args));
  },

  /**
   * Invoke a channel on the main process (returns a Promise).
   * Only channels in INVOKE_CHANNELS are accepted.
   */
  invoke(channel: string, data?: unknown): Promise<unknown> {
    if (!INVOKE_CHANNELS.includes(channel)) {
      console.warn(`[preload] Blocked unauthorized invoke channel: ${channel}`);
      return Promise.reject(new Error(`Channel not allowed: ${channel}`));
    }
    return ipcRenderer.invoke(channel, data);
  },

  /**
   * Send a one-way message to the main process.
   * Only channels in SEND_CHANNELS are accepted.
   */
  send(channel: string, data?: unknown): void {
    if (!SEND_CHANNELS.includes(channel)) {
      console.warn(`[preload] Blocked unauthorized send channel: ${channel}`);
      return;
    }
    ipcRenderer.send(channel, data);
  },
});
```

## Known Gotchas

1. **macOS Transparency:** `backgroundColor: '#00000000'` (hex with alpha) is required in some Electron versions alongside `transparent: true` — if the window shows a grey background, add `backgroundColor: '#00000000'` to the BrowserWindow options.

2. **`app.dock.hide()` in tests:** The Electron mock `app.dock` may be undefined if `dock` is not mocked. Guard with `app.dock?.hide()` (already done in index.ts).

3. **`electron-store` ESM:** Version 10+ is ESM-only. If Jest fails with `SyntaxError: Cannot use import statement`, add `electron-store` to `transformIgnorePatterns` override: `{ transformIgnorePatterns: ['node_modules/(?!(electron-store)/)'] }` in jest.config.ts.

4. **Tray icon on macOS:** A `nativeImage.createEmpty()` tray icon shows as a blank menu bar slot. Replace `assets/tray-icon.png` with a proper 16×16 template image (white with alpha channel) for the production app — template images adapt to light/dark menu bar automatically.

5. **Renderer module resolution:** `tsconfig.renderer.json` uses `"moduleResolution": "bundler"` for ES module imports. If you add a build step (e.g., esbuild), verify it handles relative imports correctly. For Phase 0 the renderer script is loaded directly as `type="module"` — no bundler needed.

6. **`require.main === module` guard:** The `if (require.main === module) { main(); }` guard in `index.ts` prevents the app lifecycle from running during Jest tests, which is why the index test can import `createWindow` without triggering `app.whenReady()`.
