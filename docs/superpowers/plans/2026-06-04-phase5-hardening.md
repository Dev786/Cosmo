```typescript
// src/main/crashRecovery.ts
import { app } from 'electron';
import Store from 'electron-store';
import fs from 'fs';
import path from 'path';
import os from 'os';

const LOG_PATH = path.join(os.homedir(), '.pixel', 'logs', 'crash.log');
const ONE_HOUR = 60 * 60 * 1000;
const MAX_RELAUNCHES_PER_HOUR = 3;

function logCrash(message: string): void {
  try {
    const line = `[${new Date().toISOString()}] ${message}\n`;
    fs.appendFileSync(LOG_PATH, line, 'utf8');
  } catch {
    // Logging must never throw — if log dir is missing we accept the silence.
  }
}

/**
 * Call once at startup. Returns true if the app should continue launching,
 * false if it has crashed too many times in the last hour (caller should quit).
 */
export function checkAndRecordLaunch(store: Store<Record<string, unknown>>): boolean {
  const now = Date.now();
  const raw = store.get('relaunches', []) as number[];
  const recent = raw.filter((ts) => now - ts < ONE_HOUR);

  if (recent.length >= MAX_RELAUNCHES_PER_HOUR) {
    logCrash(
      `Too many crashes (${recent.length} in last hour), staying down.`
    );
    return false;
  }

  store.set('relaunches', [...recent, now]);
  return true;
}

/**
 * Registers uncaughtException + render-process-gone handlers.
 * Each records a timestamp and triggers app.relaunch() + app.exit(1).
 */
export function setupCrashHandler(store: Store<Record<string, unknown>>): void {
  const onCrash = (label: string, detail: string) => {
    logCrash(`${label}: ${detail}`);
    const now = Date.now();
    const raw = store.get('relaunches', []) as number[];
    store.set('relaunches', [...raw, now]);
    app.relaunch();
    app.exit(1);
  };

  process.on('uncaughtException', (err: Error) => {
    onCrash('uncaughtException', `${err.message}\n${err.stack ?? ''}`);
  });

  (app as Electron.App & { on: Function }).on(
    'render-process-gone',
    (_event: unknown, _webContents: unknown, details: { reason: string }) => {
      onCrash('render-process-gone', details.reason);
    }
  );
}
```

```typescript
// src/main/watchers/meeting.ts

/** Process/app names that indicate a live meeting is in progress. */
const MEETING_APPS = [
  'zoom.us',
  'zoom',
  'microsoft teams',
  'facetime',
  'webex meetings',
  'webex',
  'google meet',
] as const;

/** URL substrings that indicate a browser-based meeting is the active tab. */
const MEETING_URLS = [
  'meet.google.com',
  'zoom.us/j/',
  'teams.microsoft.com',
  'chat.google.com',
] as const;

/**
 * Returns true if the given frontmost application name belongs to a meeting app.
 * Case-insensitive.
 */
export function isMeetingApp(appName: string): boolean {
  const lower = appName.toLowerCase();
  return MEETING_APPS.some((a) => lower.includes(a));
}

/**
 * Returns true if the given URL is a recognised meeting URL.
 * Pass an empty string when no browser is frontmost; returns false.
 */
export function isMeetingUrl(url: string): boolean {
  if (!url) return false;
  return MEETING_URLS.some((u) => url.includes(u));
}
```

```typescript
// src/main/core/soundQueue.ts
import path from 'path';
import { execFile, ChildProcess } from 'child_process';

export type SoundName = 'bloop' | 'chime' | 'grumble';

/**
 * Serialized afplay-backed sound player.
 * Sounds are queued and played one at a time; a missing file is skipped silently.
 * Wire sounds.enabled from config via setEnabled() before first use.
 */
export class SoundQueue {
  private enabled = true;
  private queue: string[] = [];
  private playing = false;
  private readonly soundPaths: Record<SoundName, string>;

  constructor(assetsDir: string) {
    this.soundPaths = {
      bloop: path.join(assetsDir, 'sounds', 'bloop.aiff'),
      chime: path.join(assetsDir, 'sounds', 'chime.aiff'),
      grumble: path.join(assetsDir, 'sounds', 'grumble.aiff'),
    };
  }

  play(sound: SoundName): void {
    if (!this.enabled) return;
    this.queue.push(this.soundPaths[sound]);
    if (!this.playing) this.processNext();
  }

  setEnabled(enabled: boolean): void {
    this.enabled = enabled;
  }

  private processNext(): void {
    const next = this.queue.shift();
    if (!next) {
      this.playing = false;
      return;
    }
    this.playing = true;
    const proc = execFile('afplay', [next], (_err) => {
      // Error is intentionally ignored — missing/corrupt file skipped silently.
      this.processNext();
    });
    proc.on('error', () => {
      // Handle ENOENT or permission errors without crashing.
      this.processNext();
    });
  }
}

// Singleton instance; configured from main/index.ts after reading config.
export const soundQueue = new SoundQueue(
  // __dirname is <project>/build/main/core at runtime; assets are in <project>/resources/sounds
  path.join(__dirname, '..', '..', '..', 'resources')
);
```

```javascript
// scripts/generate-sounds.js
// Run once: node scripts/generate-sounds.js
// Writes three minimal valid AIFF files (44-byte header + 100ms of silence at 22050Hz)
// to src/assets/sounds/. Replace these with real sounds at any time.

const fs = require('fs');
const path = require('path');

const OUT_DIR = path.join(__dirname, '..', 'src', 'assets', 'sounds');
fs.mkdirSync(OUT_DIR, { recursive: true });

function makeAiff(filename) {
  // AIFF-C format, 1 channel, 22050 Hz, 16-bit, ~100ms = 2205 frames
  const sampleRate = 22050;
  const numFrames = Math.floor(sampleRate * 0.1); // 100 ms
  const numChannels = 1;
  const bitsPerSample = 16;
  const bytesPerFrame = (bitsPerSample / 8) * numChannels;
  const soundDataSize = numFrames * bytesPerFrame;

  // IEEE 754 extended 80-bit representation of 22050.0
  const sampleRateBytes = Buffer.from([0x40, 0x0D, 0xAC, 0x44, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00]);

  const COMM_SIZE = 18;
  const SSND_SIZE = soundDataSize + 8; // 8 bytes offset+block
  const formSize = 4 + (8 + COMM_SIZE) + (8 + SSND_SIZE); // 'AIFF' + COMM chunk + SSND chunk

  const buf = Buffer.alloc(12 + COMM_SIZE + 8 + SSND_SIZE + 8);
  let offset = 0;

  // FORM chunk
  buf.write('FORM', offset); offset += 4;
  buf.writeUInt32BE(formSize, offset); offset += 4;
  buf.write('AIFF', offset); offset += 4;

  // COMM chunk
  buf.write('COMM', offset); offset += 4;
  buf.writeUInt32BE(COMM_SIZE, offset); offset += 4;
  buf.writeInt16BE(numChannels, offset); offset += 2;
  buf.writeUInt32BE(numFrames, offset); offset += 4;
  buf.writeInt16BE(bitsPerSample, offset); offset += 2;
  sampleRateBytes.copy(buf, offset); offset += 10;

  // SSND chunk
  buf.write('SSND', offset); offset += 4;
  buf.writeUInt32BE(SSND_SIZE, offset); offset += 4;
  buf.writeUInt32BE(0, offset); offset += 4; // offset
  buf.writeUInt32BE(0, offset); offset += 4; // blockSize
  // Sound data is all zeros (silence) — buffer was already zeroed by alloc

  const outPath = path.join(OUT_DIR, filename);
  fs.writeFileSync(outPath, buf);
  console.log(`Written: ${outPath} (${buf.length} bytes)`);
}

makeAiff('bloop.aiff');
makeAiff('chime.aiff');
makeAiff('grumble.aiff');

console.log('Done. Replace with real audio files for production.');
```

```typescript
// src/renderer/settings/settings.ts
import { IPC } from '../../shared/types';

declare const window: Window & {
  cosmo: {
    invoke: (channel: string, ...args: unknown[]) => Promise<unknown>;
    on: (channel: string, fn: (...args: unknown[]) => void) => void;
  };
};

async function init() {
  // Load config and populate all form fields
  const config = (await window.cosmo.invoke(IPC.SETTINGS_GET)) as Record<string, unknown>;
  populateForm(config);

  // Live-save on any field change
  document.addEventListener('change', (e) => {
    const target = e.target as HTMLInputElement | HTMLSelectElement;
    if (!target.name) return;
    const value = target.type === 'checkbox'
      ? (target as HTMLInputElement).checked
      : target.value;
    window.cosmo.invoke(IPC.SETTINGS_SET, { key: target.name, value });
  });

  // Update range display
  const rateInput = document.querySelector<HTMLInputElement>('[name="voice.rate"]');
  const rateVal = document.getElementById('voice-rate-val');
  rateInput?.addEventListener('input', () => {
    if (rateVal && rateInput) rateVal.textContent = rateInput.value;
  });
}

function populateForm(config: Record<string, unknown>) {
  const fields = document.querySelectorAll<HTMLInputElement | HTMLSelectElement>('[name]');
  for (const field of fields) {
    const value = getNestedValue(config, field.name);
    if (value === undefined || value === null) continue;
    if (field.type === 'checkbox') {
      (field as HTMLInputElement).checked = Boolean(value);
    } else {
      field.value = String(value);
    }
  }
  // Update range display after population
  const rateInput = document.querySelector<HTMLInputElement>('[name="voice.rate"]');
  const rateVal = document.getElementById('voice-rate-val');
  if (rateInput && rateVal) rateVal.textContent = rateInput.value;
}

function getNestedValue(obj: Record<string, unknown>, dotPath: string): unknown {
  return dotPath.split('.').reduce<unknown>((acc, key) => {
    if (acc && typeof acc === 'object') return (acc as Record<string, unknown>)[key];
    return undefined;
  }, obj);
}

document.addEventListener('DOMContentLoaded', init);
```

```javascript
// Generates a minimal placeholder .icns file (16x16 grey square).
// Replace src/assets/icon.icns with a real icon before shipping.
// Run: node scripts/generate-icon.js

const fs = require('fs');
const path = require('path');

const OUT = path.join(__dirname, '..', 'src', 'assets', 'icon.icns');
fs.mkdirSync(path.dirname(OUT), { recursive: true });

// Minimal ICNS: magic header + one 'is32' icon type (16x16 24-bit RGB)
// Each pixel is (80, 80, 100) — a dark grey-blue suitable as a placeholder
const W = 16, H = 16;
const pixelData = Buffer.alloc(W * H * 3);
for (let i = 0; i < W * H * 3; i += 3) {
  pixelData[i] = 80;     // R
  pixelData[i + 1] = 80; // G
  pixelData[i + 2] = 100;// B
}

const iconSize = 8 + pixelData.length; // chunk header (8) + pixel data
const totalSize = 8 + iconSize;        // ICNS header (8) + icon chunk

const buf = Buffer.alloc(totalSize);
buf.write('icns', 0);                    // magic
buf.writeUInt32BE(totalSize, 4);         // total file size
```

```typescript
private pausedUntil = 0;
private pauseTimer: NodeJS.Timeout | null = null;

pauseFor(ms: number): void {
  this.pausedUntil = Date.now() + ms;
  if (this.pauseTimer) clearTimeout(this.pauseTimer);
  this.pauseTimer = setTimeout(() => { this.pausedUntil = 0; }, ms);
  for (const w of this.watchers.values()) w.pause?.();
}

resume(): void {
  this.pausedUntil = 0;
  if (this.pauseTimer) { clearTimeout(this.pauseTimer); this.pauseTimer = null; }
  for (const w of this.watchers.values()) w.resume?.();
}

isPaused(): boolean {
  return Date.now() < this.pausedUntil;
}
```

```typescript
// src/main/ai/history.ts — ensure these guarantees:
// 1. Constructor loads from ~/.pixel/history.json if it exists
// 2. Every push() trims to 20 turns and writes to disk
// 3. clear() zeroes in-memory + deletes/empties the file

const HISTORY_PATH = path.join(os.homedir(), '.pixel', 'history.json');
const MAX_TURNS = 20;

export class ConversationHistory {
  private turns: ChatMessage[] = [];

  constructor() {
    try {
      const raw = fs.readFileSync(HISTORY_PATH, 'utf8');
      this.turns = JSON.parse(raw) as ChatMessage[];
    } catch {
      this.turns = []; // file doesn't exist yet — fine
    }
  }

  push(msg: ChatMessage): void {
    this.turns.push(msg);
    if (this.turns.length > MAX_TURNS) {
      this.turns = this.turns.slice(-MAX_TURNS);
    }
    this.persist();
  }

  get messages(): ChatMessage[] { return [...this.turns]; }

  clear(): void {
    this.turns = [];
    try { fs.writeFileSync(HISTORY_PATH, '[]', 'utf8'); } catch { /* ignore */ }
  }

  private persist(): void {
    try {
      fs.mkdirSync(path.dirname(HISTORY_PATH), { recursive: true });
      fs.writeFileSync(HISTORY_PATH, JSON.stringify(this.turns, null, 2), 'utf8');
    } catch (err) {
      log.warn('[history] Failed to persist:', String(err));
    }
  }
}
```
